CREATE TABLE qna(
qna_id INT AUTO_INCREMENT PRIMARY KEY,
member_id INT NOT NULL ,
qna_date date ,
anw_date date ,
qna_title VARCHAR(50),
qna_content VARCHAR(500) NOT NULL,
anw_check VARCHAR(10) DEFAULT '답변대기',
anw_content VARCHAR(500),
FOREIGN KEY(member_id) REFERENCES member(member_id));
